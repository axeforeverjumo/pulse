```python
def _resolve_effective_is_dev_task(task: Dict[str, Any], board: Dict[str, Any]) -> bool:
    """
    Resolve whether a task is a dev task.
    Explicit is_dev_task on the task wins; otherwise inherit from board.is_development.
    """
    explicit = task.get("is_dev_task")
    if explicit is not None:
        return bool(explicit)
    return bool(board.get("is_development"))
```
