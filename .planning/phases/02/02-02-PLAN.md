```python
"""
Claude Code executor service.

Dispatches dev tasks to the Claude Code CLI bridge at localhost:4201.
Claude Code handles git operations internally (clone, edit, commit, push).
No diff parsing needed -- the bridge returns structured results.
"""
from __future__ import annotations

import logging
from typing import Any, Dict, Optional

import httpx

logger = logging.getLogger(__name__)

# Generous timeout: Claude Code can take several minutes for complex tasks
_BRIDGE_TIMEOUT = 600.0  # 10 minutes


async def execute_claude_code_task(
    *,
    bridge_url: str,
    prompt: str,
    repo_url: str,
    github_token: str,
    max_budget_usd: float = 5.0,
    session_id: Optional[str] = None,
) -> Dict[str, Any]:
    """
    Send a task to the Claude Code bridge and return structured results.

    Returns dict with keys:
      - status: "completed" | "needs_continuation" | "error"
      - session_id: str (for resume)
      - result: str (Claude's text output)
      - git_log: str (recent git log)
      - git_diff: str (diff of changes)
      - git_status: str (working tree status)
    """
    action = "continue_task" if session_id else "execute_task"
    payload = {
        "action": action,
        "prompt": prompt,
        "repo_url": repo_url,
        "github_token": github_token,
        "max_budget_usd": max_budget_usd,
    }
    if session_id:
        payload["session_id"] = session_id

    async with httpx.AsyncClient(timeout=_BRIDGE_TIMEOUT) as client:
        resp = await client.post(bridge_url, json=payload)

    if resp.status_code != 200:
        error_text = resp.text[:500]
        logger.error("Claude Code bridge error (%s): %s", resp.status_code, error_text)
        return {
            "status": "error",
            "session_id": session_id,
            "result": f"Bridge error ({resp.status_code}): {error_text}",
            "git_log": "",
            "git_diff": "",
            "git_status": "",
        }

    data = resp.json()
    return {
        "status": data.get("status", "error"),
        "session_id": data.get("session_id", ""),
        "result": data.get("result", ""),
        "git_log": data.get("git_log", ""),
        "git_diff": data.get("git_diff", ""),
        "git_status": data.get("git_status", ""),
    }
```

Key design decisions:
- 10-minute timeout (Claude Code tasks can be long)
- Returns normalized dict regardless of success/failure
- Uses "continue_task" action when session_id is provided (for resume)
- No diff parsing -- Claude Code handles git internally per EXEC-05

**CRITICAL: Restructure the post-execution flow.**

After the tier routing, the existing code (lines 1712-1902) does:
1. Post agent_response as comment
2. Call _maybe_publish_agent_git_commit (parse diff from text)
3. Check completion signals
4. Handle stall detection
5. Move to QA/Done

For claude_code tier, this flow must be DIFFERENT:
- Step 2 is SKIPPED (Claude Code already committed)
- Completion is determined by `cc_status == "completed"` not by parsing "Estado: COMPLETADA" from text
- git_result is constructed from cc_result, not from _maybe_publish_agent_git_commit

The cleanest approach: after the three-way tier branch, check if tier is claude_code and handle the post-execution separately:

```python
if agent.get("tier") == "claude_code":
    # Post result as comment (will be enhanced in Plan 02-03 with block type)
    if agent_response and comment_user_id:
        await _append_agent_activity_comment(
            supabase,
            issue_id=issue_id,
            comment_user_id=comment_user_id,
            agent=agent,
            content=agent_response,
            workspace_id=task.get("workspace_id"),
            workspace_app_id=task.get("workspace_app_id"),
        )

    # Determine completion from bridge status (not from text parsing)
    task_marked_complete = (cc_status == "completed")

    # Build git_result equivalent from bridge response
    git_diff = cc_result.get("git_diff", "")
    git_result = {
        "status": "pushed" if git_diff else "no_changes",
        "detail": cc_result.get("git_log", ""),
    }

    # Handle QA/Done transition (reuse existing logic)
    if task_marked_complete:
        updates = {}
        qa_or_done = qa_id or done_id
        if qa_or_done:
            updates["state_id"] = qa_or_done
        if checklist_items:
            updates["checklist_items"] = _mark_checklist_done(checklist_items)
        if updates:
            await supabase.table("project_issues").update(updates).eq("id", issue_id).execute()

                supabase,

    return {
        "issue_id": issue_id,
        "agent_id": agent_id,
        "task_completed": task_marked_complete,
        "response": agent_response,
        "git_result": git_result,
        "claude_code_session_id": cc_session_id_out,

        "queue_recommendation": "queued" if not task_marked_complete else "completed",
```
